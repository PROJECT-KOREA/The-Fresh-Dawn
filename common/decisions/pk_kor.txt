			}

		cost = 50
		days_remove = 90

			civilian_factory_use = 2
		}

				limit = { has_idea = KOR_idea_the_poor_1 }
				swap_ideas = {

			}

				limit = { has_idea = KOR_idea_the_poor_2 }
