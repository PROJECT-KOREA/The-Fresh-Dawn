#대한민국 임시정부의 정부 수립일 기념식
news_event = {
    id = pk_news.1
    title = pk_news.1.t
    desc = pk_news.1.desc
    picture = GFX_news_event_kpg_shanghai
    
    fire_only_once = yes
    
    trigger = {
		original_tag = GGK
        has_country_flag = GGK_flag_ki_yes
		date > 1936.4.11
	}

    option = { name = pk_news.1.a }
}
